DROP DATABASE IF EXISTS cv;
CREATE DATABASE cv;

USE cv;

CREATE TABLE IF NOT EXISTS `profile` (
`id` int(5) NOT NULL AUTO_INCREMENT,
`titre` text,
`icone_titre` text,
`nom` text,
`naissance` text,
`adresse` text,
`mail` text,
`lien` text,
`commentaire` text,
primary key (id)
) ENGINE=INNODB;

CREATE TABLE IF NOT EXISTS `avatar` (
`id` int(5) NOT NULL AUTO_INCREMENT,
`id_profile` int(5),
`path_avatar` varchar(256) NOT NULL,
primary key (id),
foreign key (id_profile) references profile(id)
) ENGINE=INNODB;

CREATE TABLE IF NOT EXISTS `cv_table` (
`id` int(5) NOT NULL AUTO_INCREMENT,
`id_profile` int(5),
`patch_cv` text,
primary key (id),
foreign key (id_profile) references profile(id)
) ENGINE=INNODB;

CREATE TABLE IF NOT EXISTS `article_profile` (
`id` int(5) NOT NULL AUTO_INCREMENT,
`id_profile` int(5),
`photo_article` text,
`article` text,
primary key (id),
foreign key (id_profile) references profile(id)
) ENGINE=INNODB;

CREATE TABLE IF NOT EXISTS `categories` (
`id` int(5) NOT NULL AUTO_INCREMENT,
`libelle` varchar(256) NOT NULL,
primary key (id)
) ENGINE=INNODB;

CREATE TABLE IF NOT EXISTS `competences` (
`id` int(5) NOT NULL AUTO_INCREMENT,
`id_profile` int(5),
`id_categorie` int(5),
`sous_categorie` text,
`image_sous_categorie` text,
`details` text,
primary key (id),
foreign key (id_profile) references profile(id),
foreign key (id_categorie) references categories(id)
) ENGINE=INNODB;

CREATE TABLE IF NOT EXISTS `parcours` (
`id` int(5) NOT NULL AUTO_INCREMENT,
`titre` text,
`date_debut` date,
`date_fin` date,
`description` text,
primary key (id)
) ENGINE=INNODB;

CREATE TABLE IF NOT EXISTS `missions` (
`id` int(5) NOT NULL AUTO_INCREMENT,
`id_parcours` int(5),
`mission` text,
primary key (id),
foreign key (id_parcours) references parcours(id)
) ENGINE=INNODB;
